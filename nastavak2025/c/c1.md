import java.util.Scanner;

public class c1 {
    public static void main(String[] args) {
        int a, b, zbir, proizvod, razlika, celobrojniKolicnik, celobrojnoDeljenje;
        double realniKolicnik;

        Scanner sc = new Scanner(System.in);

        System.out.print(" a = ");
        a = sc.nextInt();
        System.out.print(" b = ");
        b = sc.nextInt();

        zbir = a + b;
        System.out.println("\nЗбир: " + zbir);

        razlika = a - b;
        System.out.println("\nРазлика: " + razlika);

        proizvod = a * b;
        System.out.println("\nПроизвод: " + proizvod);

        celobrojniKolicnik = a / b;
        System.out.println("\nЦелобројни количник: " + celobrojniKolicnik);

        realniKolicnik = (double) a / b;
        System.out.println("\nРеални количник: " + realniKolicnik);

        celobrojnoDeljenje = a % b;
        System.out.println("\nЦелобројно дељење: " + celobrojnoDeljenje);

        sc.close();
    }
}
